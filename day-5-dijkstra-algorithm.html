dijkstra

In computer networks it is very important for us to route data packets efficiently between the nodes.So the fundamental problem is finding shortest path between the nodes which will minimize the latency and maximizes the bandwidth.

What is Dijkstra's Algorithm?

Dijkstra Algorithm helps us find the shortest path from source node to all other nodes in a weighted graph.It is very widely used in OSPF. These algorithm maintains priority queue so that we can greedily select the closest node and update the distance to its neighbors.

the code for the following will be:-

    
        #include<'iostream'>
        #include<'vector'>
		 #include<'queue'>
		 #include<'climits'>
		 #include<'utility'>
		 #include<'unistd.h'>
		 
		 using namespace std;
		 
		 class Graph{
		      int numNodes;
					vector<'vector<'pair<'int,int'>>'>adjList;
					
					public:
					   Graph(int n) : numNodes(n),adjList(n) {}
						 
						 void addEdge(int src,int dest,int weight)
						 {
						    adjList[src].push_back({dest,weight});
								adjList[dest].push_back({src,weight});
							}
							
							vector<'int'> dijkstra(int start)
							{
							   vector<'int'> dist(numNodes,INT_MAX);
								 priority_queue<'pair<'int,int''>,vector<'pair<'int,int''>>,greater<'pair<'int,int''>>> pq;
							
						  dist[start] = 0;
							pq.push({0,start});
							while(!pq.empty())
              {
							  int currentDist = pq.top().first;
								int currentNode = pq.top().second;
								pq.pop();
								if(currentDist > dist[currentNode])
								{
								  continue;
								}
								for(auto &neighbor : adjList[currentNode])
							  {
								  int nextNode = neighbor.first;
									int edgeWeight = neighbor.second;
									if(dist[nextNode] > dist[currentNode] + edgeWeight)
									{
									   dist[nextNode] = dist[currentNode] + edgeWeight;
										 pq.push({dist[nextNode] ,nextNode});
									}
							}
					}
	};
